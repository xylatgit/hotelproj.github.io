hotel_df %>% 
  select(is_canceled, adr) %>% 
  group_by(is_canceled, adr) %>% 
  summarize(
    count = n()
  ) %>% 
  filter(count > 10, is_canceled == 1) %>% 
  drop_na() %>% 
  mutate(
    fv = lm(count ~ adr, .) %>%  fitted.values()
  ) %>% 
  plot_ly(x = ~adr, y = ~count, mode = "markers") %>%
  add_markers(y = ~count) %>% 
  add_trace(x = ~adr, y = ~fv, mode = "lines") %>% 
  layout(title = "Average Daily Rate (ADR) and Cancellation", 
         xaxis = list(title = "ADR"), 
         yaxis = list(title = "Cancellation Count"),
         showlegend = F)
